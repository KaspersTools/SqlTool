34 struct comp {
35 bool operator()(const std::string &lhs, const std::string &rhs) const {
36 // On non Windows OS, use the function "strcasecmp" in #include <strings.h>
37 return strcasecmp(lhs.c_str(), rhs.c_str()) == 0;
38 }
39 };
40 struct hash {
41 std::size_t operator()(std::string str) const {
42 for (std::size_t index = 0; index < str.size(); ++index) {
43 auto ch = static_cast<unsigned char>(str[index]);
44 str[index] = static_cast<unsigned char>(std::tolower(ch));
45 }
46 return std::hash<std::string>{}(str);
47 }
48 };
49 };
